#!/usr/bin/env bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
REPO_ROOT="$(cd "$SCRIPT_DIR/../.." && pwd)"

FIXTURES="$SCRIPT_DIR/fixtures"
PASS=0
FAIL=0

assert_pass() {
  local desc="$1"; shift
  if "$@" > /dev/null 2>&1; then
    echo "PASS: $desc"
    ((PASS++)) || true
  else
    echo "FAIL: $desc"
    ((FAIL++)) || true
  fi
}

assert_fail() {
  local desc="$1"; shift
  local expected_error="$1"; shift
  local output
  if output=$("$@" 2>&1); then
    echo "FAIL: $desc (expected failure, got success)"
    ((FAIL++)) || true
  elif echo "$output" | grep -q "$expected_error"; then
    echo "PASS: $desc"
    ((PASS++)) || true
  else
    echo "FAIL: $desc (expected '$expected_error' in output, got: $output)"
    ((FAIL++)) || true
  fi
}

TMPDIR=$(mktemp -d)
trap 'rm -rf "$TMPDIR"' EXIT

### Step 2: Add supervision validation to do_schema()

In `scripts/validate-plan`, add supervision validation to the `do_schema()` function, after the existing top-level field checks (around line 87, after the `success_criteria` loop but before the phase loop).

```bash
local has_supervision
has_supervision=$(echo "$json" | jq 'has("supervision")')
if [[ "$has_supervision" == "true" ]]; then
  local supervision_type
  supervision_type=$(echo "$json" | jq -r '.supervision | type')
  if [[ "$supervision_type" != "object" ]]; then
    err "ERROR: invalid_supervision_type: supervision must be an object, got $supervision_type"
  else
    local valid_supervision_keys="orchestrator_poll_seconds dispatcher_poll_seconds max_intervention_attempts"
    local supervision_keys
    supervision_keys=$(echo "$json" | jq -r '.supervision | keys[]')
    for key in $supervision_keys; do
      if [[ " $valid_supervision_keys " != *" $key "* ]]; then
        err "ERROR: unknown_supervision_field: '$key'"
      fi
    done

    for field in orchestrator_poll_seconds dispatcher_poll_seconds max_intervention_attempts; do
      local field_val
      field_val=$(echo "$json" | jq ".supervision.$field // null")
      if [[ "$field_val" != "null" ]]; then
        local field_type
        field_type=$(echo "$json" | jq -r ".supervision.$field | type")
        if [[ "$field_type" != "number" ]]; then
          err "ERROR: invalid_supervision_field: supervision.$field must be a positive integer, got $field_type"
        else
          local is_positive_int
          is_positive_int=$(echo "$json" | jq ".supervision.$field | (. > 0 and . == (. | floor))")
          if [[ "$is_positive_int" != "true" ]]; then
            err "ERROR: invalid_supervision_field: supervision.$field must be a positive integer, got $field_val"
          fi
        fi
      fi
    done
  fi
fi
```
